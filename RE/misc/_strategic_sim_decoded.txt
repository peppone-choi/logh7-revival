/**
 * Strategic galaxy AI simulation core — the "macro" layer above npc-ai.mjs.
 *
 * npc-ai.mjs drives TACTICAL ships inside one battle grid (x/z fire, move, retreat). This module is the
 * STRATEGIC layer: NPC fleet commanders decide which neighbouring star system to advance into, defend, or
 * reinforce toward — moving fleets across the galaxy and conquering systems turn by turn. It reuses the
 * authoritative world-state API only (seedSystems is the boot step; this module then drives
 * upsertFleet / moveFleet / conquerSystem / setSystemOwner). It NEVER emits wire frames — the auth-server
 * wiring layer turns the per-tick result into 0x0325 unit-table broadcasts. Pure + deterministic given an
 * explicit tick seed (mulberry32, no Math.random / no I/O / no clock) so the whole thing is unit-testable.
 *
 * ───────────────────────────────────────────────────────────────────────────────────────────────────────
 * DATA GRADE (do NOT promote to original):
 *   P0  galaxy faction / cx,cy / is_corridor          (content/galaxy.json, recovered)
 *   P0  commander 8-ability stats                      (content/character-roster.json, real values)
 *   P3  adjacency graph topology (cx,cy nearest-k)     — galaxy.json has NO edge list; this is an
 *                                                          INFERRED euclidean-proximity graph, not canon routes.
 *   P3  fleet strength / supply / reinforcement rate   — gameplay tuning seeds, unrelated to LOGH VII balance.
 *   P3  battle / scoring / decision formulas           — gameplay tuning seeds.
 * The strategic balance of the original game is unknown; every numeric tuning constant below is a P3 seed.
 * ───────────────────────────────────────────────────────────────────────────────────────────────────────
 */

import { readFileSync } from 'node:fs';
import { behaviorProfile } from './logh7-npc-ai.mjs';

// Fleet id space. Player grid-slot ids and NPC tactical seeds (0x0a000000) must not collide, so strategic
// fleets live in their own high block. (P3 — id allocation convention, not canon.)
export const STRAT_FLEET_BASE = 0x0b000000;

// Canon capitals / front anchors (galaxy.json verified: ヴァルハラ empire cx288 cy647; ヴァンフリート alliance
// corridor cx203 cy352; イゼルローン empire corridor; フェザーン neutral corridor — left unassigned). The
// alliance capital borrows its corridor front anchor (alliance has no distinct capital system in the
// recovered galaxy). Fallback to the highest-degree node is deterministic — see resolveCapital. (P0 names.)
export const DEFAULT_CAPITALS = { empire: 'ヴァルハラ', alliance: 'ヴァンフリート' };

// Numeric faction id for the wire (0x0325 unit record FACTION u16 slot). String factions ('empire'/'alliance')
// drive ownership comparisons against world systems; the world-state fleet entity carries the numeric form.
export const FACTION_WIRE = { empire: 1, alliance: 2, neutral: 0 };

// ── tuning seeds (all P3) ───────────────────────────────────────────────────────────────────────────────
const TUNING = {
  baseStrength: 1000, // starting fleet combat power (P3)
  baseSupply: 600, // starting reserve pool drawn on to reinforce (P3)
  reinforceRate: 80, // strength regained per reinforce tick, before tochi bonus (P3)
  battleNoise: 0.3, // ±15% random swing band in resolveStrategicBattle (P3)
  survivorFloor: 0.15, // fraction of strength a defeated fleet keeps, scaled by caution (P3)
};

/**
 * Deterministic PRNG (mulberry32). Same seed → same stream. Used everywhere instead of Math.random so the
 * whole simulation is reproducible from (baseSeed, tickNo, fleetId).
 * @param {number} seed
 * @returns {() => number} a function returning floats in [0,1)
 */
export function mulberry32(seed) {
  let a = seed >>> 0;
  return function next() {
    a |= 0;
    a = (a + 0x6d2b79f5) | 0;
    let t = Math.imul(a ^ (a >>> 15), 1 | a);
    t = (t + Math.imul(t ^ (t >>> 7), 61 | t)) ^ t;
    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
  };
}

/** Read a system's faction from either a world-state system ({owner}) or a content/galaxy record ({faction}). */
function systemFaction(sys) {
  return sys.owner ?? sys.faction ?? 'neutral';
}

/** Read a system's cx/cy from either {map:{cx,cy}} or flat {cx,cy}. Returns null when absent. */
function systemCoords(sys) {
  const cx = sys.map?.cx ?? sys.cx ?? null;
  const cy = sys.map?.cy ?? sys.cy ?? null;
  if (typeof cx === 'number' && typeof cy === 'number') return { cx, cy };
  return null;
}

/** A system's display name from either {name} (world-state) or {system}/{name} (galaxy record). */
function systemName(sys) {
  return sys.name ?? sys.system ?? null;
}

/**
 * Build a DETERMINISTIC adjacency graph from cx/cy proximity. content/galaxy.json carries no edge list, so
 * "neighbouring system" is approximated by euclidean nearest-k. Corridor systems (イゼルローン / フェザーン /
 * ヴァンフリート …) get a larger reach (×corridorScale) so they act as the long front-line connectors they
 * are in canon. Symmetrised (a∈N(b) ⇔ b∈N(a)). Graph is INFERRED topology (P3) — never canon routes.
 *
 * @param {Array<object>} systems world-state listSystems() OR content galaxy records (both shapes accepted)
 * @param {{ kNearest?:number, maxDist?:number, corridorScale?:number }} [opts]
 * @returns {{ nodes: Map<string,{name,faction,isCorridor,cx,cy}>, neighbors: Map<string,string[]>, distance:(a:string,b:string)=>number }}
 */
export function buildStrategicGraph(systems, { kNearest = 4, maxDist = 120, corridorScale = 1.5 } = {}) {
  /** @type {Map<string,{name:string,faction:string,isCorridor:boolean,cx:number|null,cy:number|null}>} */
  const nodes = new Map();
  for (const sys of systems) {
    const name = systemName(sys);
    if (name == null) continue;
    const coords = systemCoords(sys);
    nodes.set(name, {
      name,
      faction: systemFaction(sys),
      isCorridor: Boolean(sys.isCorridor ?? sys.is_corridor),
      cx: coords?.cx ?? null,
      cy: coords?.cy ?? null,
    });
  }

  const distCache = new Map();
  const distKey = (a, b) => (a < b ? `${a}${b}` : `${b}${a}`);
  const distance = (a, b) => {
    const key = distKey(a, b);
    if (distCache.has(key)) return distCache.get(key);
    const na = nodes.get(a);
    const nb = nodes.get(b);
    let d = Infinity;
    if (na && nb && na.cx != null && nb.cx != null) {
      d = Math.hypot(na.cx - nb.cx, na.cy - nb.cy);
    }
    distCache.set(key, d);
    return d;
  };

  // Pick nearest-k within reach for each node. Tie-break by name (lexicographic) for determinism.
  const names = [...nodes.keys()].sort();
  /** @type {Map<string,Set<string>>} */
  const adj = new Map(names.map((n) => [n, new Set()]));
  for (const a of names) {
    const na = nodes.get(a);
    if (na.cx == null) continue; // isolated (no coords) — safe: stays empty
    const reach = na.isCorridor ? maxDist * corridorScale : maxDist;
    const ranked = names
      .filter((b) => b !== a && nodes.get(b).cx != null)
      .map((b) => ({ b, d: distance(a, b) }))
      .filter((e) => e.d <= reach)
      .sort((x, y) => (x.d - y.d) || (x.b < y.b ? -1 : x.b > y.b ? 1 : 0));
    for (const { b } of ranked.slice(0, kNearest)) {
      adj.get(a).add(b);
      adj.get(b).add(a); // symmetrise (undirected)
    }
  }

  /** @type {Map<string,string[]>} */
  const neighbors = new Map();
  for (const a of names) {
    neighbors.set(a, [...adj.get(a)].sort());
  }
  return { nodes, neighbors, distance };
}

/** Breadth-first hop distance from `start` to the nearest node whose faction is in `enemyFactions`. ∞ if none. */
function hopsToEnemy(graph, start, ownFaction) {
  const seen = new Set([start]);
  let frontier = [start];
  let hops = 0;
  while (frontier.length) {
    const next = [];
    for (const cur of frontier) {
      const node = graph.nodes.get(cur);
      if (node && node.faction !== ownFaction && node.faction !== 'neutral') {
        return hops;
      }
      for (const nb of graph.neighbors.get(cur) ?? []) {
        if (!seen.has(nb)) {
          seen.add(nb);
          next.push(nb);
        }
      }
    }
    frontier = next;
    hops += 1;
  }
  return Infinity;
}

/** Resolve a faction's capital: the named system if owned/present, else its highest-degree owned node. */
function resolveCapital(graph, faction, preferred) {
  if (preferred && graph.nodes.has(preferred) && graph.nodes.get(preferred).faction === faction) {
    return preferred;
  }
  const owned = [...graph.nodes.values()].filter((n) => n.faction === faction);
  if (owned.length === 0) return preferred ?? null;
  // highest neighbour count, tie-break by name (deterministic)
  owned.sort((a, b) => {
    const da = (graph.neighbors.get(a.name) ?? []).length;
    const db = (graph.neighbors.get(b.name) ?? []).length;
    return (db - da) || (a.name < b.name ? -1 : a.name > b.name ? 1 : 0);
  });
  return owned[0].name;
}

let cachedRoster = null;
/** Load the canon roster once (content/character-roster.json). Returns the characters array. */
function loadRoster() {
  if (cachedRoster) return cachedRoster;
  try {
    const raw = readFileSync(new URL('../../content/character-roster.json', import.meta.url), 'utf8');
    cachedRoster = JSON.parse(raw).characters ?? [];
  } catch {
    cachedRoster = [];
  }
  return cachedRoster;
}

/**
 * Deterministically pick the commanders for a faction: roster characters of that faction WITH known stats,
 * ordered by 統率 tochi desc then name (so the strongest leaders take the front fleets). Returns up to `count`.
 * @returns {Array<{ charId:number, stats:object, name:string }>}
 */
function pickCommanders(roster, faction, count) {
  const pool = roster
    .filter((c) => c.faction === faction && c.stats_known && c.stats)
    .map((c, i) => ({ charId: i, raw: c, stats: c.stats, name: c.name_romaji ?? c.name_ja ?? c.id }))
    .sort((a, b) => (b.stats.tochi ?? 0) - (a.stats.tochi ?? 0) || (a.name < b.name ? -1 : a.name > b.name ? 1 : 0));
  // charId를 진영 접두(FACTION_WIRE*0x1000)로 전역 유일화 — 이전엔 진영별 1..n이라 empire 1==alliance 1
  // 충돌(향후 commander→character 조인 시 동일인 오인). 0x0325 COMMANDER 슬롯값으로도 나간다(P3 슬롯).
  const base = (FACTION_WIRE[faction] ?? 0) * 0x1000;
  return pool.slice(0, count).map((e, i) => ({ charId: base + i + 1, stats: e.stats, name: e.name }));
}

/**
 * Map a system name to a strategic cell (row*100+col). Deterministic: derived from the node's index in the
 * graph's sorted node list. Used to give the world-state fleet entity a position the 0x0325 unit table can
 * carry. (P3 — cell layout is a convenience encoding, not a canon sector grid.)
 */
export function systemNameToCell(name, graph) {
  const names = [...graph.nodes.keys()].sort();
  const idx = names.indexOf(name);
  if (idx < 0) return 0;
  const row = Math.floor(idx / 10);
  const col = idx % 10;
  return row * 100 + col;
}

/**
 * Seed the strategic fleets at boot. For each playable faction (empire/alliance), the owned systems closest
 * to the enemy front get one fleet each (the rest fall back to the capital). Each fleet's commander is a
 * canon officer of that faction (pickCommanders, strongest first) and its behaviour comes from that officer's
 * real stats via behaviorProfile (reused, unmodified, from npc-ai). Also creates the matching world-state
 * fleet entity (upsertFleet) so the wire unit table has it. (Strength/supply VALUES are P3 seeds.)
 *
 * @param {object} worldState createWorldState()
 * @param {ReturnType<typeof buildStrategicGraph>} graph
 * @param {{ seed?:number, fleetsPerFaction?:number, capitals?:object, factions?:string[], roster?:Array }} [opts]
 * @returns {{ fleets: object[], byFaction: Map<string, object[]>, capitals: object }}
 */
export function seedStrategicFleets(worldState, graph, {
  seed = 1,
  fleetsPerFaction = 6,
  capitals = DEFAULT_CAPITALS,
  factions = ['empire', 'alliance'],
  roster = null,
} = {}) {
  const rosterChars = roster ?? loadRoster();
  const fleets = [];
  const byFaction = new Map();
  const resolvedCapitals = {};
  let nextId = 0;

  for (const faction of factions) {
    const capital = resolveCapital(graph, faction, capitals[faction]);
    resolvedCapitals[faction] = capital;
    // owned systems sorted by front proximity (fewest hops to an enemy system), tie-break by name.
    const owned = [...graph.nodes.values()]
      .filter((n) => n.faction === faction)
      .map((n) => ({ name: n.name, hops: hopsToEnemy(graph, n.name, faction) }))
      .sort((a, b) => (a.hops - b.hops) || (a.name < b.name ? -1 : a.name > b.name ? 1 : 0));
    const commanders = pickCommanders(rosterChars, faction, fleetsPerFaction);

    const placed = [];
    for (let i = 0; i < fleetsPerFaction; i += 1) {
      const home = owned[i]?.name ?? capital ?? owned[0]?.name;
      if (!home) break; // faction has no owned systems — skip safely
      const cmd = commanders[i] ?? commanders[commanders.length - 1] ?? { charId: 0, stats: {}, name: 'unknown' };
      const profile = behaviorProfile(cmd.stats);
      const id = STRAT_FLEET_BASE + nextId;
      nextId += 1;
      const fleet = {
        id,
        faction,
        commander: cmd.charId,
        commanderName: cmd.name,
        homeSystem: home,
        system: home,
        strength: TUNING.baseStrength,
        supply: TUNING.baseSupply,
        order: 'hold',
        profile,
        stats: cmd.stats,
        targetSystem: null,
        moving: false,
      };
      fleets.push(fleet);
      placed.push(fleet);
      // wire entity: position via cell, numeric faction, commander id. Strength etc. stay in the sim Map.
      worldState.upsertFleet({
        id,
        owner: 0,
        faction: FACTION_WIRE[faction] ?? 0,
        commander: cmd.charId,
        cell: systemNameToCell(home, graph),
        supply: TUNING.baseSupply,
        mapSection: 0,
      });
    }
    byFaction.set(faction, placed);
  }

  return { fleets, byFaction, capitals: resolvedCapitals };
}

/**
 * Resolve a strategic battle between two fleets (PURE). Effective power = strength × command bonus (指揮
 * shiki) × attack bonus (攻撃 kogeki) × a small deterministic noise band. The higher effective power wins;
 * the loser is reduced to a caution-scaled survivor floor, the winner takes proportional losses. All
 * constants are P3 tuning seeds (formula mirrors npc-ai's damageMul shape, not original balance).
 *
 * @param {object} attacker StratFleet
 * @param {object} defender StratFleet
 * @param {() => number} rng deterministic PRNG
 * @returns {{ winner:object, loser:object, attackerLoss:number, defenderLoss:number, attackerPower:number, defenderPower:number }}
 */
export function resolveStrategicBattle(attacker, defender, rng) {
  const effPower = (f, r) => {
    const command = (f.profile?.command ?? 0.5);
    const aggression = (f.profile?.aggression ?? 0.5);
    const noise = (1 - TUNING.battleNoise / 2) + TUNING.battleNoise * r; // 0.85..1.15 with default
    return f.strength * (0.8 + command * 0.5) * (0.9 + aggression * 0.2) * noise;
  };
  const attackerPower = effPower(attacker, rng());
  const defenderPower = effPower(defender, rng());
  const attackerWins = attackerPower >= defenderPower;
  const winner = attackerWins ? attacker : defender;
  const loser = attackerWins ? defender : attacker;

  // loser survivor fraction scales with its own caution (defensive commanders preserve more).
  const survivorFrac = TUNING.survivorFloor + (loser.profile?.caution ?? 0.3) * 0.2;
  const loserBefore = loser.strength;
  loser.strength = Math.max(0, Math.round(loser.strength * survivorFrac));
  // winner loss proportional to how close the loser's power was (a hard fight costs more).
  const winnerPower = attackerWins ? attackerPower : defenderPower;
  const loserPower = attackerWins ? defenderPower : attackerPower;
  const lossFrac = Math.min(0.6, 0.15 + 0.45 * (loserPower / (winnerPower || 1)));
  const winnerBefore = winner.strength;
  winner.strength = Math.max(1, Math.round(winner.strength * (1 - lossFrac)));

  return {
    winner,
    loser,
    attackerLoss: attackerWins ? winnerBefore - winner.strength : loserBefore - loser.strength,
    defenderLoss: attackerWins ? loserBefore - loser.strength : winnerBefore - winner.strength,
    attackerPower,
    defenderPower,
  };
}

/**
 * Decide one NPC fleet's strategic order (PURE — no mutation). Priority: defend exposed home → advance on an
 * enemy/neutral neighbour → reinforce when crippled → hold. Scoring weights the commander's personality
 * (aggression=攻撃, caution=防御, mobility=機動 reach, command=指揮) plus 統率 tochi easing the supply-line
 * penalty (a high-統率 leader projects power further). All weights are P3 seeds.
 *
 * @returns {{ order:'defend'|'advance'|'reinforce'|'hold', target?:string|null }}
 */
export function decideStrategicOrder(fleet, worldState, graph, simState, rng) {
  const here = fleet.system;
  const neighbors = graph.neighbors.get(here) ?? [];
  const ownFaction = fleet.faction;
  const tochi = (fleet.stats?.tochi ?? 80) / 120; // 0..~1 supply-line reach
  const profile = fleet.profile ?? {};

  // is an enemy fleet adjacent (incursion)?
  const enemyAdjacent = neighbors.some((nb) => {
    for (const f of simState.fleetsById.values()) {
      if (f.faction !== ownFaction && f.system === nb && f.strength > 0) return true;
    }
    return false;
  });

  // 3. reinforce: crippled → fall back / rebuild.
  const integrity = fleet.strength / (TUNING.baseStrength || 1);
  if (integrity <= (profile.retreatBelow ?? 0.3) && fleet.supply > 0) {
    return { order: 'reinforce', target: fleet.homeSystem };
  }

  // gather enemy/neutral neighbour candidates and score them.
  const candidates = [];
  for (const nb of neighbors) {
    const node = graph.nodes.get(nb);
    if (!node) continue;
    const sys = worldState.getSystem(nb);
    const owner = sys ? sys.owner : node.faction;
    if (owner === ownFaction) continue; // own territory — not a target
    const isNeutral = owner === 'neutral';
    const dist = graph.distance(here, nb);
    // defended? an enemy fleet sitting on the target.
    let defenderStrength = 0;
    for (const f of simState.fleetsById.values()) {
      if (f.system === nb && f.faction !== ownFaction && f.strength > 0) defenderStrength += f.strength;
    }
    let score = 0;
    score += isNeutral ? 40 : 60; // enemy worth more than neutral
    score += (profile.aggression ?? 0.5) * 60; // bold commanders push harder
    score -= dist * (0.4 - tochi * 0.25); // distance penalty, eased by 統率
    score -= defenderStrength / (TUNING.baseStrength || 1) * (1 - (profile.aggression ?? 0.5)) * 50; // cautious avoid strong defenders
    if (isNeutral) score += (profile.caution ?? 0.5) * 20; // cautious commanders prefer soft neutral targets
    score += rng() * 10; // small deterministic jitter
    candidates.push({ target: nb, score, defenderStrength, isNeutral });
  }

  // 1. defend: exposed home + cautious + an enemy is adjacent → hold the line.
  if (enemyAdjacent && (profile.caution ?? 0.5) >= 0.45) {
    return { order: 'defend', target: here };
  }

  // 2. advance on the best-scoring target.
  if (candidates.length) {
    candidates.sort((a, b) => (b.score - a.score) || (a.target < b.target ? -1 : a.target > b.target ? 1 : 0));
    return { order: 'advance', target: candidates[0].target };
  }

  // 4. hold (no enemy/neutral neighbour — rear system): accumulate supply.
  return { order: 'hold', target: null };
}

/**
 * One authoritative strategic tick. Each NPC fleet decides an order and the world-state is mutated through
 * its existing authority API (moveFleet / conquerSystem). Deterministic: PRNG per fleet is mulberry32(seed
 * ^ fleetId), and the seed for tick N is baseSeed ^ (N*0x9e3779b1). worldRelay-independent (no broadcast
 * here — the wiring layer does that). Returns a structured diff for the caller to broadcast/trace.
 *
 * @param {object} worldState createWorldState()
 * @param {ReturnType<typeof buildStrategicGraph>} graph
 * @param {{ fleetsById: Map<number,object>, baseSeed:number }} simState
 * @param {{ seed?:number, tickNo?:number }} [opts]
 * @returns {{ moves:object[], conquests:object[], battles:object[], reinforcements:object[], economy:object[], decisions:object[] }}
 */
export function strategicTick(worldState, graph, simState, { seed, tickNo = 0 } = {}) {
  const baseSeed = (seed ?? simState.baseSeed ?? 1) >>> 0;
  const tickSeed = (baseSeed ^ (Math.imul(tickNo, 0x9e3779b1))) >>> 0;
  const moves = [];
  const conquests = [];
  const battles = [];
  const reinforcements = [];
  const economy = [];
  const decisions = [];

  // deterministic fleet order (by id) so the result sequence is reproducible.
  const fleetList = [...simState.fleetsById.values()].sort((a, b) => a.id - b.id);

  for (const fleet of fleetList) {
    if (fleet.strength <= 0) continue; // destroyed fleet skips (rebuild handled via reinforce on survivors)
    const rng = mulberry32((tickSeed ^ fleet.id) >>> 0);
    const decision = decideStrategicOrder(fleet, worldState, graph, simState, rng);
    decisions.push({ fleetId: fleet.id, order: decision.order, target: decision.target });

    if (decision.order === 'reinforce') {
      const tochiBonus = 1 + (fleet.stats?.tochi ?? 80) / 240; // 統率 speeds recovery (P3)
      const amount = Math.min(fleet.supply, Math.round(TUNING.reinforceRate * tochiBonus));
      fleet.supply -= amount;
      fleet.strength += amount;
      fleet.order = 'reinforce';
      // fall back toward home if not already there.
      if (fleet.system !== fleet.homeSystem && (graph.neighbors.get(fleet.system) ?? []).includes(fleet.homeSystem)) {
        applyMove(worldState, graph, simState, fleet, fleet.homeSystem, moves);
      }
      reinforcements.push({ fleetId: fleet.id, system: fleet.system, amount });
      continue;
    }

    if (decision.order === 'advance' && decision.target) {
      const target = decision.target;
      const sys = worldState.getSystem(target);
      const targetOwner = sys ? sys.owner : graph.nodes.get(target)?.faction;
      // is the target defended by an enemy fleet?
      const defenders = fleetList.filter((f) => f.system === target && f.faction !== fleet.faction && f.strength > 0);
      if (defenders.length === 0) {
        // undefended enemy/neutral → move in, conquer.
        applyMove(worldState, graph, simState, fleet, target, moves);
        if (targetOwner !== fleet.faction) {
          worldState.conquerSystem(target, fleet.faction);
          if (graph.nodes.get(target)) graph.nodes.get(target).faction = fleet.faction; // keep graph in sync
          conquests.push({ system: target, from: targetOwner ?? 'neutral', to: fleet.faction });
          economy.push({ system: target, owner: fleet.faction });
        }
        fleet.order = 'advance';
      } else {
        // defended → resolve battle against the strongest defender.
        const defender = defenders.sort((a, b) => b.strength - a.strength)[0];
        const result = resolveStrategicBattle(fleet, defender, rng);
        const won = result.winner === fleet;
        battles.push({
          system: target,
          attacker: fleet.faction,
          defender: defender.faction,
          winner: result.winner.faction,
          attackerLoss: result.attackerLoss,
          defenderLoss: result.defenderLoss,
        });
        syncFleetWire(worldState, fleet);
        syncFleetWire(worldState, defender);
        if (won) {
          // attacker takes the system; broken defender retreats home (or stays to rebuild if it lives nearby).
          applyMove(worldState, graph, simState, fleet, target, moves);
          worldState.conquerSystem(target, fleet.faction);
          if (graph.nodes.get(target)) graph.nodes.get(target).faction = fleet.faction;
          conquests.push({ system: target, from: targetOwner ?? defender.faction, to: fleet.faction });
          economy.push({ system: target, owner: fleet.faction });
          if (defender.strength > 0 && defender.system === target) {
            const fallback = (graph.neighbors.get(target) ?? []).includes(defender.homeSystem)
              ? defender.homeSystem
              : (graph.neighbors.get(target) ?? []).find((nb) => worldState.getSystem(nb)?.owner === defender.faction);
            if (fallback) applyMove(worldState, graph, simState, defender, fallback, moves);
          }
        }
        // loser==attacker → stays put (repulsed), already had strength cut by resolveStrategicBattle.
      }
      continue;
    }

    if (decision.order === 'defend') {
      fleet.order = 'defend';
      continue;
    }

    // hold → accumulate supply.
    fleet.order = 'hold';
    fleet.supply += 20; // P3 rear-system supply accrual
  }

  return { moves, conquests, battles, reinforcements, economy, decisions };
}

/** Move a fleet to `to`: update the sim StratFleet.system AND the world-state fleet cell (kept in lockstep). */
function applyMove(worldState, graph, simState, fleet, to, moves) {
  const from = fleet.system;
  if (from === to) return;
  fleet.system = to;
  fleet.targetSystem = null;
  fleet.moving = false;
  worldState.moveFleet(fleet.id, systemNameToCell(to, graph));
  moves.push({ fleetId: fleet.id, from, to });
}

/** Push a fleet's mutable sim values (supply) back onto its world-state wire entity. */
function syncFleetWire(worldState, fleet) {
  const ent = worldState.getFleet(fleet.id);
  if (ent) {
    ent.supply = fleet.supply;
  }
}

/**
 * Factory: seed the strategic fleets and return a tick-driver + snapshot. The auth-server wiring calls this
 * one function (after seedSystems + buildStrategicGraph) and then `tick(n)` on its interval.
 *
 * @param {object} worldState createWorldState()
 * @param {ReturnType<typeof buildStrategicGraph>} graph
 * @param {{ seed?:number, fleetsPerFaction?:number, capitals?:object, factions?:string[], roster?:Array }} [opts]
 * @returns {{ simState:{fleetsById:Map<number,object>, baseSeed:number}, tick:(tickNo?:number)=>object, snapshot:()=>{fleets:object[]}, seedResult:object }}
 */
export function createStrategicSim(worldState, graph, opts = {}) {
  const baseSeed = (opts.seed ?? 1) >>> 0;
  const seedResult = seedStrategicFleets(worldState, graph, opts);
  /** @type {Map<number,object>} */
  const fleetsById = new Map(seedResult.fleets.map((f) => [f.id, f]));
  const simState = { fleetsById, baseSeed };
  // 月間(30게임일) 연령효과 드리프트 주기(틱→게임시간 매핑은 캐논 미수록 → P3 튜닝). 매 ticksPerMonth 틱마다
  // worldState.applyMonthlyAgeDrift 호출 → 무유저 갤럭시에서 인물 능력치가 성장/쇠퇴(abilities/age 시드 캐릭터만).
  const ticksPerMonth = (opts.ticksPerMonth ?? 30) >>> 0;
  return {
    simState,
    seedResult,
    tick(tickNo = 0) {
      const result = strategicTick(worldState, graph, simState, { seed: baseSeed, tickNo });
      if (ticksPerMonth > 0 && tickNo > 0 && tickNo % ticksPerMonth === 0) {
        // 월간 경계: 인물 능력치 드리프트 + 자동진급(사다리 #1). 무유저 갤럭시서 사령관단이 시간에 따라 진화.
        if (typeof worldState.applyMonthlyAgeDrift === 'function') result.ageDrift = worldState.applyMonthlyAgeDrift();
        if (typeof worldState.runMonthlyPromotions === 'function') result.promotions = worldState.runMonthlyPromotions();
      }
      return result;
    },
    snapshot() {
      return { fleets: [...fleetsById.values()].map((f) => ({ ...f })) };
    },
  };
}
